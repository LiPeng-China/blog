## 11、等式

```latex
% 使用 \& 使 = 左对齐

\begin{aligned}
    (f,K^{'}_{x}y+K^{''}_{x}y)_{F}
    &=(f^{'}+f^{''},K^{'}_{x}y+K^{''}_{x}y)_{F}\\
    &=(f^{'},K^{'}_{x}y)_{F}+(f^{''},K^{''}_{x}y)_{F}+(f^{'},K^{''}_{x}y)_{F}+(f^{''},K^{'}_{x}y){F}\\
    &=(f^{'},K^{'}_{x}y)_{F}+(f^{''},K^{''}_{x}y)_{F}\\
    &=(f^{'}(x),y)_{Y}+(f^{''}(x),y)_{Y}\\
    &=(f^{'}(x)+f^{''}(x),y)_{Y}\\
    &=(f(x),y)_{Y}\\
    &=(f,K_{x}y)_{F}
\end{aligned}
```

$$
\begin{aligned}
    (f,K^{'}_{x}y+K^{''}_{x}y)_{F}
    &=(f^{'}+f^{''},K^{'}_{x}y+K^{''}_{x}y)_{F}\\
    &=(f^{'},K^{'}_{x}y)_{F}+(f^{''},K^{''}_{x}y)_{F}+(f^{'},K^{''}_{x}y)_{F}+(f^{''},K^{'}_{x}y){F}\\
    &=(f^{'},K^{'}_{x}y)_{F}+(f^{''},K^{''}_{x}y)_{F}\\
    &=(f^{'}(x),y)_{Y}+(f^{''}(x),y)_{Y}\\
    &=(f^{'}(x)+f^{''}(x),y)_{Y}\\
    &=(f(x),y)_{Y}\\
    &=(f,K_{x}y)_{F}
\end{aligned}
$$
